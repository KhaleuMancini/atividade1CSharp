SQL 4 - Busca todos os funcionários no banco

"select	funcionarios.nome as 'Nome',
	email as 'Email',
	tel as 'Telefone',
	pis as 'PIS',
	cargos.nome as 'Cargo',
	setores.nome as 'Setor',
	admissao as 'Data de admissão',
	salario as 'Salário',
	desconto as 'Desconto de Funcionário'
from funcionarios
inner join cargos on idcargofk = cargo.id
inner join setores on idsetorfk = setores.id
order by funcionarios.nome;"
